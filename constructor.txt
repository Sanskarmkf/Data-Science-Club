4. Predict the output of the following code:

 	Dog() {
 		name = "Unknown";
 		age = 0;
 	}
 	void display() {
 		System.out.println("Dog's Name: " + name + " Age: " + age);
 	}
 }
 public class Main {
 	public static void main(String[] args) {
 		Dog myDog = new Dog(); // Calls the default constructor
 		myDog.display();
 	}
